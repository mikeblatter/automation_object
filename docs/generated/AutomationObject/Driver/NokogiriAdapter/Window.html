def back
  raise UnableToNavigateBackward.new if @position == 0
  @position -= 1

  self.make_request(@history.at(@position))
end

def forward
  raise UnableToNavigateForward.new if @position >= @history.length - 2
  @position += 1

  self.make_request(@history.at(@position))
end
