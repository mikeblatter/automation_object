def load(type, name)
  case type
  when :screen
    composite.use(name)
  when :modal
    composite.current_screen.use(name)
  when :element, :element_array, :element_hash
    return composite.current_screen.get(type, name)
  else
    raise AutomationObject::State::UndefinedLoadTypeError
  end
end
