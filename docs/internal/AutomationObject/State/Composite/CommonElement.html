def method_hooks
  return @method_hooks if @method_hooks

  @method_hooks = {}
  blue_prints.method_hooks.each do |key, blue_prints|
    @method_hooks[key] = Hook.new(blue_prints,
                                  driver,
                                  key,
                                  self,
                                  location + "[#{key}]")
  end

  @method_hooks
end
