def changes
  changes = []
  method_hooks.each_value do |hook|
    changes += hook.changes
  end

  changes.uniq.compact
end

def method_to_container(name)
  method_hooks.each do |method_name, hook|
    next unless hook.changes.include?(name)

    return method_name
  end

  nil
end
