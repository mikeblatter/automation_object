def method_missing(method, *args, &block)
  return super if Screen.methods.include?(method)

  # Attempt to load screen if composite object contains that child
  @state.utilize if @subject.to_h.include?(method)

  super
end

def element(name)
  name = name.to_sym
  raise AutomationObject::Dsl::Error::ElementDoesNotExistError.new(name) unless @subject.to_h.include?(name)

  @state.utilize
  @subject.send(name)
end
