def initialize(driver, blue_prints)
  self.driver = driver
  self.blue_prints = blue_prints

  self.state = State.new(self.driver, self.blue_prints)
  @subject = Dsl.new(self.blue_prints, self.state)

  AutomationObject::Framework.singleton = self
end
